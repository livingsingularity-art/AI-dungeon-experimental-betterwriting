const modifier = (text) => {
  text = AutoCards("input", text);
  text = LocalizedLanguages("input", text);
  const newText = text.replace(/\bQw\b(?:\s+([^\n]+))?/g, (match, p1) => {
    if (p1) {
      return `[Describe: ${p1.trim()}]\n`;
    }
    return `[Describe:]`;
  });
  return { text: newText };
}

// Don't modify this part
modifier(text)
